server {
        listen 80;

        index index.php index.html index.htm index.nginx-debian.html;

        server_name homework.xntvps.ru hwnt.ru;

        charset utf-8;

        location / {
                try_files /public/$uri /public/$uri/ /index.php;

        location = /index.php {
                include snippets/fastcgi-php.conf;
